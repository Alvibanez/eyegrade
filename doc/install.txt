# Clonar el repositorio
git clone <user>@varpa.it.uc3m.es:~jaf/lib/www/eyegrade/git

# Installar paquetes requeridos
sudo aptitude install python-pygame python-opencv libcv4 libavformat52
sudo aptitude install python-dev libtre5 libtre-dev

# Instalar los bindings de python de libtre:
wget http://laurikari.net/tre/tre-0.8.0.tar.gz
tar xzvf tre-0.8.0.tar.gz
cd tre-0.8.0/python/
python setup.py build
sudo python setup.py install

# Fichero de configuración
# camera-dev: 0 si es la primera cámara
# si el ordenador tiene ya una integrada, camera-dev: 1
cp <camgrade-DIR>/git/doc/camgrade.cfg.sample ~/.camgrade.cfg

cd <eyegrade-DIR>/git/src
python eyegrade/eyegrade.py fichero.eye --id-list students.txt
